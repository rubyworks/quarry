      def delete(loc, src, dest, opts)
        if File.exist?(dest)
          how = 'delete'
          begin
            rm(dest)
          rescue Errno::ENOTEMPTY
            how = 'skip'
          end
        else
          how = 'missing'
        end
        return how, dest
      end
