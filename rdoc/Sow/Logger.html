    def report_fixes(marker='FIXME:')
      glist = check_for_fixes(marker)
      unless glist.empty?
        puts "\nYou need to fix the occurances of '#{marker}' in the following files:\n\n"
        glist.each do |file|
          puts "      #{file}"
        end
        puts
      end
    end
