# File lib/sow/metadata.rb, line 32
    def method_missing(s, *a, &b)
      s = s.to_s
      if s =~ /=$/
        s = s.chomp('=')
        self[s] = a[0]
      else
        s = s.chomp('?')
        self[s]
      end
    end
